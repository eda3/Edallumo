18/// JSONファイルを読み込む
19///
20/// 指定されたパスのJSONファイルを読み込み、指定された型にデシリアライズします。
21///
22/// # 引数
23/// * `path` - 読み込むファイルのパス
24///
25/// # 戻り値
26/// `Result<T>` - デシリアライズされたデータ
27#[allow(dead_code)]
28pub fn read_json_file<T>(path: impl AsRef<Path>) -> Result<T>
29where
30    T: for<'de> serde::Deserialize<'de>,
31{
32    let path = path.as_ref();
33    let mut file = File::open(path).map_err(|e| {
34        AppError::FileNotFound(format!(
35            "ファイルを開けませんでした: {} - {}",
36            path.display(),
37            e
38        ))
39    })?;
40
41    let mut contents = String::new();
42    file.read_to_string(&mut contents).map_err(|e| {
43        AppError::Io(io::Error::new(
44            io::ErrorKind::Other,
45            format!(
46                "ファイルの読み込みに失敗しました: {} - {}",
47                path.display(),
48                e
49            ),
50        ))
51    })?;
52
53    serde_json::from_str(&contents).map_err(|e| {
54        AppError::Json(de::Error::custom(format!(
55            "JSONの解析に失敗しました: {} - {}",
56            path.display(),
57            e
58        )))
59    })
60}

62/// JSONデータをファイルに書き込む
63///
64/// 指定されたデータをJSONに変換し、指定されたパスのファイルに書き込みます。
65///
66/// # 引数
67/// * `path` - 書き込み先ファイルのパス
68/// * `data` - 書き込むデータ
69///
70/// # 戻り値
71/// `Result<()>` - 書き込み結果
72#[allow(dead_code)]
73pub fn write_json_file<T>(path: impl AsRef<Path>, data: &T) -> Result<()>
74where
75    T: serde::Serialize,
76{
77    let path = path.as_ref();
78    let json = serde_json::to_string_pretty(data).map_err(|e| {
79        AppError::Json(ser::Error::custom(format!(
80            "JSONへの変換に失敗しました: {e}"
81        )))
82    })?;
83
84    let mut file = File::create(path).map_err(|e| {
85        AppError::Io(io::Error::new(
86            io::ErrorKind::Other,
87            format!(
88                "ファイルの作成に失敗しました: {path_display} - {e}",
89                path_display = path.display()
90            ),
91        ))
92    })?;
93
94    file.write_all(json.as_bytes()).map_err(|e| {
95        AppError::Io(io::Error::new(
96            io::ErrorKind::Other,
97            format!(
98                "ファイルの書き込みに失敗しました: {path_display} - {e}",
99                path_display = path.display()
100            ),
101        ))
102    })?;
103
104    Ok(())
105}

107/// 複数のJSONファイルを読み込む
108///
109/// 指定されたディレクトリ内のJSONファイルをすべて読み込み、指定された型にデシリアライズします。
110///
111/// # 引数
112/// * `dir` - 読み込むディレクトリのパス
113/// * `extension` - 読み込むファイルの拡張子（デフォルトは "json"）
114///
115/// # 戻り値
116/// `Result<Vec<T>>` - デシリアライズされたデータのベクター
117#[allow(dead_code)]
118pub fn read_all_json_files<T>(dir: impl AsRef<Path>, extension: Option<&str>) -> Result<Vec<T>>
119where
120    T: for<'de> serde::Deserialize<'de>,
121{
122    let dir_path = dir.as_ref();
123    let mut result = Vec::new();
124
125    for entry in fs::read_dir(dir_path).map_err(|e| {
126        AppError::FileNotFound(format!(
127            "ディレクトリを読み込めませんでした: {} - {}",
128            dir_path.display(),
129            e
130        ))
131    })? {
132        let entry = entry.map_err(|e| {
133            AppError::Io(io::Error::new(
134                io::ErrorKind::Other,
135                format!("ディレクトリエントリの読み込みに失敗しました: {e}"),
136            ))
137        })?;
138
139        let path = entry.path();
140        if path.is_file()
141            && path
142                .extension()
143                .map_or(false, |ext| ext == extension.unwrap_or("json"))
144        {
145            let data = read_json_file(&path)?;
146            result.push(data);
147        }
148    }
149
150    Ok(result)
151}

238/// 括弧内が外と同じ場合、括弧を除去する
239///
240/// 「2HS(2HS)」のように括弧内と外が同じ文言の場合、括弧を含む部分を削除します。
241/// 例: 「2HS(2HS)」→「2HS」、「足払い(2D)」→「足払い(2D)」（変更なし）
242///
243/// # 引数
244/// * `input` - 処理する入力文字列
245///
246/// # 戻り値
247/// `String` - 処理された文字列
248#[allow(dead_code)]
249pub fn remove_redundant_brackets(input: &str) -> String {
250    // 括弧内のコンテンツを抽出
251    if let Some(start) = input.find('(') {
252        if let Some(end) = input.find(')') {
253            if end > start {
254                // 括弧外の内容
255                let outside = &input[0..start].trim();
256                // 括弧内の内容
257                let inside = &input[start + 1..end].trim();
258
259                // 括弧内と外が同じ場合は括弧部分を除去
260                if outside == inside {
261                    return (*outside).to_string();
262                }
263            }
264        }
265    }
266
267    // 条件に合わない場合は元の文字列を返す
268    input.to_string()
269}

289/// 入力に一致する技を検索する
290///
291/// # 引数
292/// * `moves` - 検索対象の技情報のベクター
293/// * `aliases` - 技エイリアス情報のベクター
294/// * `query` - 検索クエリ
295///
296/// # 戻り値
297/// `Option<MoveInfo>` - 一致した技情報（見つからなかった場合は None）
298#[allow(dead_code)]
299pub fn find_move(moves: &[MoveInfo], aliases: &[MoveAliases], query: &str) -> Option<MoveInfo> {
300    let normalized_query = normalize_input(query);
301
302    // 入力コマンドで検索
303    for move_info in moves {
304        if normalize_input(&move_info.input) == normalized_query {
305            return Some(move_info.clone());
306        }
307    }
308
309    // 技名で検索
310    for move_info in moves {
311        if normalize_move_name(&move_info.name) == normalize_move_name(&normalized_query) {
312            return Some(move_info.clone());
313        }
314    }
315
316    // エイリアスで検索
317    for alias in aliases {
318        if alias
319            .aliases
320            .iter()
321            .any(|a| normalize_input(a) == normalized_query)
322        {
323            // エイリアスが見つかったら、対応する技を返す
324            for move_info in moves {
325                if normalize_input(&move_info.input) == normalize_input(&alias.input) {
326                    return Some(move_info.clone());
327                }
328            }
329        }
330    }
331
332    None
333}

335/// ファイルのMD5ハッシュを計算する
336///
337/// 指定されたファイルのMD5ハッシュを計算します。
338///
339/// # 引数
340/// * `path` - ファイルのパス
341///
342/// # 戻り値
343/// `Result<String>` - MD5ハッシュ文字列
344#[allow(dead_code)]
345pub fn calculate_file_hash(path: impl AsRef<Path>) -> Result<String> {
346    let path = path.as_ref();
347    let mut file = File::open(path).map_err(|e| {
348        AppError::FileNotFound(format!(
349            "ファイルを開けませんでした: {} - {}",
350            path.display(),
351            e
352        ))
353    })?;
354
355    let mut buffer = Vec::new();
356    file.read_to_end(&mut buffer).map_err(|e| {
357        AppError::Io(io::Error::new(
358            io::ErrorKind::Other,
359            format!(
360                "ファイルの読み込みに失敗しました: {} - {}",
361                path.display(),
362                e
363            ),
364        ))
365    })?;
366
367    let mut hasher = Md5::new();
368    hasher.update(&buffer);
369    let result = hasher.finalize();
370
371    Ok(format!("{result:x}"))
372}

375///

403            )

405        );
406    }

411/// 文字列を指定した長さに切り詰める
412///
413/// 文字列が指定した長さを超える場合、切り詰めて末尾に「...」を追加します。
414///
415/// # 引数
416/// * `s` - 切り詰める文字列
417/// * `max_len` - 最大長
418///
419/// # 戻り値
420/// `String` - 切り詰められた文字列
421#[allow(dead_code)]
422pub fn truncate_string(s: &str, max_len: usize) -> String {
423    if s.len() <= max_len {
424        s.to_string()
425    } else {
426        let mut result = s.chars().take(max_len - 3).collect::<String>();
427        result.push_str("...");
428        result
429    }
430}
