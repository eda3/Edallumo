6use poise::serenity_prelude as serenity;
7use std::io;
8use thiserror::Error;
9
10/// アプリケーション全体で使用するエラー型
11#[derive(Error, Debug)]
12pub enum AppError {
13    /// IOエラー
14    #[error("IO エラー: {0}")]
15    Io(#[from] io::Error),
16
17    /// JSON解析エラー
18    #[error("JSON 解析エラー: {0}")]
19    Json(#[from] serde_json::Error),
20
21    /// ファイル未検出エラー
22    #[error("ファイルが見つかりません: {0}")]
23    FileNotFound(String),
24
25    /// キャラクター未検出エラー
26    #[error("キャラクターが見つかりません: {0}")]
27    CharacterNotFound(String),
28
29    /// 技未検出エラー
30    #[error("技が見つかりません: {0}")]
31    MoveNotFound(String),
32
33    /// Discord APIエラー
34    #[error("Discord API エラー: {0}")]
35    Discord(String),
36
37    /// Serenity APIエラー
38    #[error("Serenity API エラー: {0}")]
39    Serenity(#[from] serenity::Error),
40
41    /// 設定エラー
42    #[error("設定エラー: {0}")]
43    Config(String),
44
45    /// データ処理エラー
46    #[error("データ処理エラー: {0}")]
47    DataProcessing(String),
48
49    /// その他のエラー
50    #[error("エラー: {0}")]
51    Other(String),
52}
53
54// `Box<dyn std::error::Error>型からAppError型への変換
55impl From<Box<dyn std::error::Error>> for AppError {
56    fn from(error: Box<dyn std::error::Error>) -> Self {
57        AppError::Other(error.to_string())
58    }
59}
60
61/// `結果型の別名定義（アプリケーション全体で使用）`
62pub type Result<T> = std::result::Result<T, AppError>;

64#[cfg(test)]
65mod tests {
66    use super::*;
67    use std::io::{Error as IoError, ErrorKind};
68
69    #[test]
70    fn test_app_error_io() {
71        let io_error = IoError::new(ErrorKind::NotFound, "テスト用IOエラー");
72        let app_error = AppError::from(io_error);
73
74        if let AppError::Io(e) = app_error {
75            assert_eq!(e.kind(), ErrorKind::NotFound);
76        } else {
77            panic!("変換されたエラーの型が正しくありません");
78        }
79    }
80
81    #[test]
82    fn test_app_error_file_not_found() {
83        let error_message = "test.json";
84        let app_error = AppError::FileNotFound(error_message.to_string());
85
86        let error_string = app_error.to_string();
87        assert!(error_string.contains(error_message));
88    }
89
90    #[test]
91    fn test_app_error_character_not_found() {
92        let character = "存在しない_キャラ";
93        let app_error = AppError::CharacterNotFound(character.to_string());
94
95        let error_string = app_error.to_string();
96        assert!(error_string.contains(character));

99    #[test]
100    fn test_app_error_move_not_found() {

108    #[test]
