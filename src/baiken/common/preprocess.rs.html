709/// 与えられたJSON文字列に対して共通の置換処理を実施する関数
710///
711/// # 概要
712/// 入力されたJSON文字列内の特定パターンを、所定の日本語表記へ変換する。  
713/// フレームデータ等の文字列の正規化に利用する。
714///
715/// # 引数
716/// * `json` - 置換前のJSON文字列
717///
718/// # 戻り値
719/// 置換処理後のJSON文字列
720///
721/// # 例
722/// ```rust,no_run
723/// use your_crate::preprocess_json;
724/// let input = r#"{"key": "c.S", "other": "f.S"}"#.to_string();
725/// let output = preprocess_json(input);
726/// println!("{}", output);
727/// ```
728pub fn preprocess_json(json: String) -> String {
729    let mut result = json;
730
731    // 正規表現による置換
732    result = RE_C_S.replace_all(&result, r#""近S""#).to_string();
733    result = RE_F_S.replace_all(&result, r#""遠S""#).to_string();
734    result = RE_J_MOVE.replace_all(&result, r#""j$1""#).to_string();
735    result = RE_HKD_MOVE
736        .replace_all(&result, r#"強制ダウン"#)
737        .to_string();
738    result = RE_KD_MOVE.replace_all(&result, r#"ダウン"#).to_string();
739
740    // マップを使用した置換
741    for (pattern, replacement) in REPLACEMENT_MAP.iter() {
742        result = result.replace(pattern, replacement);
743    }
744
745    result
746}
