13/// データフォルダ "data" 存在チェック関数である。  
14/// 引数：`init_check` - 初期チェックか否かの真偽値  
15/// 戻り値：チェック成功時 None / エラー発生時エラーメッセージ (Some(String)) を返す。
16pub async fn data_folder_exists(init_check: bool) -> Option<String> {
17    // "data" フォルダ存在確認
18    if Path::new("data").exists() {
19        None
20    } else {
21        // "data" フォルダ未存在エラー用メッセージ
22        let error_msg = "Error: The 'data' folder does not exist.\nDownload and import the 'data' folder from:\nhttps://github.com/yakiimoninja/baiken.".to_string();
23
24        if init_check {
25            // 初期チェック時、エラーメッセージをコンソール出力しパニック
26            println!();
27            panic!("{}", error_msg.red());
28        } else {
29            // Discord などで出力するためエラーメッセージを返す
30            Some(error_msg)
31        }
32    }
33}

35/// nicknames.json 存在および正当性チェック関数である。  
36/// 引数：`init_check` - 初期チェックか否かの真偽値  
37/// 戻り値：正常時 None / エラー発生時エラーメッセージ (Some(String)) を返す。
38pub async fn nicknames_json_exists(init_check: bool) -> Option<String> {
39    // nicknames.json ファイル読み込み
40    let Ok(data_from_file) = fs::read_to_string("data/nicknames.json") else {
41        let error_msg = "Error: Failed to read 'nicknames.json' file.\nDownload and import the `data` folder from:\nhttps://github.com/yakiimoninja/baiken.".to_string();
42
43        if init_check {
44            println!();
45            panic!("{}", error_msg.red());
46        } else {
47            return Some(error_msg);
48        }
49    };
50
51    // JSON デシリアライズ試行
52    if serde_json::from_str::<Vec<Nicknames>>(&data_from_file).is_ok() {
53        println!("{}", "Successfully read 'nicknames.json' file.".green());
54        None
55    } else {
56        // nicknames.json 正当性エラー用メッセージ
57        let error_msg = "Error: Failed to deserialize 'nicknames.json' file.\nDownload and import the `data` folder from:\nhttps://github.com/yakiimoninja/baiken.".to_string();
58
59        if init_check {
60            println!();
61            panic!("{}", error_msg.red());
62        } else {
63            Some(error_msg)
64        }
65    }
66}

81                panic!("{}", error_msg.red());
82            } else {
83                return Some(error_msg);
84            }
85        }
86    }

141bitflags! {
142    /// チェックオプションフラグ
143    ///
144    /// 各種チェック機能の有効・無効を指定するためのビットフラグ
145    #[derive(Default)]
146    pub struct CheckOptions: u8 {
147        /// データフォルダの存在チェック
148        const DATA_FOLDER = 0b00001;
149        /// ニックネームJSONの存在チェック
150        const NICKNAMES_JSON = 0b00010;
151        /// キャラクターフォルダの存在チェック
152        const CHARACTER_FOLDERS = 0b00100;
153        /// キャラクターJSONの存在チェック
154        const CHARACTER_JSONS = 0b01000;
155        /// キャラクター画像の存在チェック
156        const CHARACTER_IMAGES = 0b10000;
157        /// すべてのチェックを実行
158        const ALL = Self::DATA_FOLDER.bits() | Self::NICKNAMES_JSON.bits() |
159                   Self::CHARACTER_FOLDERS.bits() | Self::CHARACTER_JSONS.bits() |
160                   Self::CHARACTER_IMAGES.bits();
161    }
162}

164/// アダプティブチェック関数
165///
166/// 指定されたオプションに基づいて、必要なチェックのみを実行する
167///
168/// # 引数
169/// * `ctx` - Discordのコンテキスト
170/// * `options` - 実行するチェックを指定するオプションフラグ
171///
172/// 戻り値：全チェック成功時 Ok(()) / 失敗時 Err("Failed `adaptive_check`")
173pub async fn adaptive_check(ctx: Context<'_>, options: CheckOptions) -> Result<()> {
174    if options.contains(CheckOptions::DATA_FOLDER) {
175        // Checking if data folder exists
176        if let Some(error_msg) = data_folder_exists(false).await {
177            if let Err(e) = ctx.say(&error_msg.replace('\'', "`")).await {
178                println!("Failed to send message: {e}");
179            }
180            println!();
181            panic!("{}", error_msg.replace('\n', " ").red());
182        }
183    }
184    if options.contains(CheckOptions::NICKNAMES_JSON) {
185        // Checking if nicknames.json exists
186        if let Some(error_msg) = nicknames_json_exists(false).await {
187            if let Err(e) = ctx.say(&error_msg.replace('\'', "`")).await {
188                println!("Failed to send message: {e}");
189            }
190            println!();
191            panic!("{}", error_msg.replace('\n', " ").red());
192        }
193    }
194    if options.contains(CheckOptions::CHARACTER_FOLDERS) {
195        // Checking if character folders exist
196        if let Some(error_msg) = character_folders_exist(false).await {
197            if let Err(e) = ctx.say(&error_msg.replace('\'', "`")).await {
198                println!("Failed to send message: {e}");
199            }
200            println!();
201            panic!("{}", error_msg.replace('\n', " ").red());
202        }
203    }
204    if options.contains(CheckOptions::CHARACTER_JSONS) {
205        // Checking if character jsons exist
206        if let Some(error_msg) = character_jsons_exist(false).await {
207            if let Err(e) = ctx.say(&error_msg.replace('\'', "`")).await {
208                println!("Failed to send message: {e}");
209            }
210            println!();
211            panic!("{}", error_msg.replace('\n', " ").red());
212        }
213    }
214    if options.contains(CheckOptions::CHARACTER_IMAGES) {
215        // Checking if character images exist
216        if let Some(error_msg) = character_images_exist(false).await {
217            if let Err(e) = ctx.say(&error_msg.replace('\'', "`")).await {
218                println!("Failed to send message: {e}");
219            }
220            println!();
221            panic!("{}", error_msg.replace('\n', " ").red());
222        }
223    }
224
225    Ok(())
226}

228/// データディレクトリの検証を行う
229///
230/// 指定されたデータディレクトリが存在し、必要なサブディレクトリやファイルが
231/// 含まれているかを確認します。
232///
233/// # 引数
234/// * `data_dir` - データディレクトリのパス
235///
236/// # 戻り値
237/// `Result<()>` - 成功時は `Ok(())`, 失敗時はエラー
238pub fn validate_data_dir(data_dir: &str) -> Result<()> {
239    println!("{}", "データディレクトリの検証を開始します...".cyan());
240
241    // データディレクトリの存在確認
242    let data_path = Path::new(data_dir);
243    if !data_path.exists() {
244        return Err(AppError::FileNotFound(format!(
245            "データディレクトリが見つかりません: {data_dir}"
246        )));
247    }
248
249    if !data_path.is_dir() {
250        return Err(AppError::Config(format!(
251            "指定されたパスはディレクトリではありません: {data_dir}"
252        )));
253    }
254
255    println!(
256        "{}",
257        format!("データディレクトリを確認しました: {data_dir}").green()
258    );
259
260    // 必要なサブディレクトリの存在確認（例として一部のキャラクター）
261    let essential_chars = ["Sol_Badguy", "Ky_Kiske", "May"];
262    for char_name in &essential_chars {
263        let char_dir = data_path.join(char_name);
264        if !char_dir.exists() || !char_dir.is_dir() {
265            return Err(AppError::Config(format!(
266                "必要なキャラクターディレクトリが見つかりません: {char_name}"
267            )));
268        }
269    }
270
271    println!("{}", "基本キャラクターデータの存在を確認しました".green());
272    println!("{}", "データディレクトリの検証が完了しました".green());
273
274    Ok(())
275}

277/// キャラクターフォルダの整合性チェック
278///
279/// すべてのキャラクター用フォルダが存在するかチェックする
280///
281/// # 引数
282/// * `data_dir` - データディレクトリへのパス
283///
284/// # 戻り値
285/// `Result<()>` - 成功時は (), 失敗時はエラー
286#[allow(dead_code)]
287pub fn check_character_folders(data_dir: &str) -> Result<()> {
288    println!("{}", "キャラクターフォルダの検証を開始します...".cyan());
289
290    let data_path = Path::new(data_dir);
291    if !data_path.exists() {
292        return Err(AppError::FileNotFound(format!(
293            "データディレクトリが見つかりません: {data_dir}"
294        )));
295    }
296
297    // CHARS配列からキャラクターフォルダを確認
298    for char_name in &crate::CHARS {
299        let char_dir = data_path.join(char_name);
300        if !char_dir.exists() || !char_dir.is_dir() {
301            println!(
302                "{}",
303                format!(
304                    "警告: キャラクターフォルダが見つかりません: {}",
305                    char_dir.display()
306                )
307                .yellow()
308            );
309            // 警告のみ出して続行
310        }
311    }
312
313    println!("{}", "キャラクターフォルダの検証が完了しました".green());
314
315    Ok(())
316}

318/// データ整合性チェック
319///
320/// 全キャラクターデータの整合性を検証する
321///
322/// # 引数
323/// * `data_dir` - データディレクトリへのパス
324#[allow(dead_code)]
325pub fn check_data_integrity(data_dir: &str) {
326    println!("{}", "データ整合性の検証を開始します...".cyan());
327
328    let data_path = Path::new(data_dir);
329
330    // 各キャラクターのJSONファイルをサンプリングチェック
331    for char_name in &["Sol_Badguy", "Ky_Kiske", "May"] {
332        let char_dir = data_path.join(char_name);
333        if !char_dir.exists() || !char_dir.is_dir() {
334            continue; // このキャラクターはスキップ
335        }
336
337        // character.json の整合性チェック
338        let char_data_path = char_dir.join("character.json");
339        if char_data_path.exists() && char_data_path.is_file() {
340            match fs::read_to_string(&char_data_path) {
341                Ok(content) => {
342                    if let Err(e) = serde_json::from_str::<crate::models::CharInfo>(&content) {
343                        println!(
344                            "{}",
345                            format!(
346                                "警告: キャラクターデータの解析に失敗しました: {} - {e}",
347                                char_data_path.display()
348                            )
349                            .yellow()
350                        );
351                    }
352                }
353                Err(e) => {
354                    println!(
355                        "{}",
356                        format!(
357                            "警告: キャラクターデータの読み込みに失敗しました: {} - {e}",
358                            char_data_path.display()
359                        )
360                        .yellow()
361                    );
362                }
363            }
364        }
365
366        // moves.json の整合性チェック
367        let moves_data_path = char_dir.join("moves.json");
368        if moves_data_path.exists() && moves_data_path.is_file() {
369            match fs::read_to_string(&moves_data_path) {
370                Ok(content) => {
371                    if let Err(e) = serde_json::from_str::<Vec<crate::models::MoveInfo>>(&content) {
372                        println!(
373                            "{}",
374                            format!(
375                                "警告: 技データの解析に失敗しました: {} - {e}",
376                                moves_data_path.display()
377                            )
378                            .yellow()
379                        );
380                    }
381                }
382                Err(e) => {
383                    println!(
384                        "{}",
385                        format!(
386                            "警告: 技データの読み込みに失敗しました: {} - {e}",
387                            moves_data_path.display()
388                        )
389                        .yellow()
390                    );
391                }
392            }
393        }
394    }
395
396    println!("{}", "データ整合性の検証が完了しました".green());
397}
