15/// テスト用の一時ディレクトリ構造を作成する
16///
17/// テスト用の一時ディレクトリとその中にキャラクターデータ用のサブディレクトリを作成します。
18///
19/// # 戻り値
20/// `(TempDir, PathBuf)` - 一時ディレクトリハンドルとそのパス
21#[cfg(test)]
22pub fn create_test_dir_structure() -> (TempDir, PathBuf) {
23    let temp_dir = TempDir::new().expect("一時ディレクトリの作成に失敗しました");
24    let temp_path = temp_dir.path().to_path_buf();
25
26    fs::create_dir_all(temp_path.join("Sol_Badguy"))
27        .expect("Sol_Badguyディレクトリの作成に失敗しました");
28    fs::create_dir_all(temp_path.join("Ky_Kiske"))
29        .expect("Ky_Kiskeディレクトリの作成に失敗しました");
30    fs::create_dir_all(temp_path.join("May")).expect("Mayディレクトリの作成に失敗しました");
31
32    (temp_dir, temp_path)
33}

176/// モック用のDiscordコンテキストを作成する
177///
178/// テスト用のモックDiscordコンテキストを作成します。
179/// 実際のAPIは呼び出さず、内部処理のテストに使用します。
180///
181/// このモックは実際のコンテキストと同じインターフェースを持ちますが、
182/// 常に成功するレスポンスを返します。
183#[cfg(test)]
184pub struct _MockDiscordContext {
185    pub messages: Vec<String>,
186    pub embeds: Vec<String>,
187}
188
189#[cfg(test)]
190impl _MockDiscordContext {
191    /// 新しいモックコンテキストを作成する
192    ///
193    /// # 戻り値
194    /// `MockDiscordContext` - 新しいモックコンテキスト
195    pub fn new() -> Self {
196        Self {
197            messages: Vec::new(),
198            embeds: Vec::new(),
199        }
200    }
201
202    /// メッセージを送信する（モック）
203    ///
204    /// # 引数
205    /// * `content` - メッセージ内容
206    ///
207    /// # 戻り値
208    /// `Result<(), ()>` - 常に `Ok(())`
209    pub fn send_message(&mut self, content: &str) -> Result<(), ()> {
210        self.messages.push(content.to_string());
211        Ok(())
212    }
213
214    /// 埋め込みメッセージを送信する（モック）
215    ///
216    /// # 引数
217    /// * `content` - 埋め込みメッセージ内容
218    ///
219    /// # 戻り値
220    /// `Result<(), ()>` - 常に `Ok(())`
221    pub fn send_embed(&mut self, content: &str) -> Result<(), ()> {
222        self.embeds.push(content.to_string());
223        Ok(())
224    }
225}
