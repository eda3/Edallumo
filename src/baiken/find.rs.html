20/// キャラクター名を検索し、該当する正式なキャラクター名を返却する非同期関数
21///
22/// # 概要
23/// nicknames.json ファイルを読み込み、ユーザーの入力文字列と比較して  
24/// 一致または部分一致するキャラクターの正式名称を返却する。
25///
26/// # 引数
27/// * `character` - ユーザー入力のキャラクター名またはニックネーム
28///
29/// # 戻り値
30/// 正式なキャラクター名を含む `Result<String>` を返す
31pub async fn find_character(character: &String) -> Result<String> {
32    // nicknames.json ファイル読み込み　結果：JSON文字列取得
33    let data_from_file = fs::read_to_string("data/nicknames.json").map_err(|e| {
34        AppError::FileNotFound(format!("nicknames.jsonの読み込みに失敗しました: {e}"))
35    })?;
36
37    // JSON文字列を Nicknames 構造体のベクターへデシリアライズ　結果：vec_nicknames
38    let vec_nicknames =
39        serde_json::from_str::<Vec<Nicknames>>(&data_from_file).map_err(AppError::Json)?;
40
41    // 各キャラクターエントリ走査　結果：該当エントリ検出時に正式名称返却
42    for x_nicknames in &vec_nicknames {
43        // 各ニックネーム走査　結果：入力文字列と完全一致すれば正式名称返却
44        for y_nicknames in &x_nicknames.nicknames {
45            if y_nicknames.to_lowercase() == character.to_lowercase().trim() {
46                return Ok(x_nicknames.character.clone());
47            }
48        }
49    }
50
51    // キャラクター名の部分一致走査　結果：入力文字列が正式名称の一部に含まれていれば返却
52    for x_nicknames in &vec_nicknames {
53        if x_nicknames
54            .character
55            .to_lowercase()
56            .replace('-', "")
57            .contains(&character.to_lowercase())
58            || x_nicknames
59                .character
60                .to_lowercase()
61                .contains(&character.to_lowercase())
62        {
63            return Ok(x_nicknames.character.clone());
64        }
65    }
66
67    // "all" の場合の例外処理　結果：空文字列返却
68    if character.trim().to_lowercase() == "all".to_lowercase() {
69        return Ok(String::new());
70    }
71
72    // キャラクター未検出時エラーメッセージ作成　結果：エラー返却
73    let error_msg = "Character `".to_owned() + character + "` was not found!";
74    Err(AppError::CharacterNotFound(error_msg))
75}

77/// 技のインデックスを検索し、該当する技のインデックスを返却する非同期関数
78///
79/// # 概要
80/// ```指定されたキャラクターの技情報（moves_info）から、
81/// ユーザー入力に対応する技のインデックスを検索する。
82/// 入力がエイリアスであれば、実際の技入力に変換する。```
83///
84/// # 引数
85/// * `character_arg_altered` - 正式なキャラクター名
86/// * `character_move` - ユーザー入力の技名、入力、またはエイリアス
87/// * `moves_info` - キャラクターの技情報のスライス
88///
89/// # 戻り値
90/// 該当技のインデックスを含む `Result<usize>` を返す
91pub async fn find_move_index(
92    character_arg_altered: &String,
93    mut character_move: String,
94    moves_info: &[MoveInfo],
95) -> Result<usize> {
96    // 対象キャラクターの aliases.json のパス生成　結果：aliases_path
97    let aliases_path = "data/".to_owned() + character_arg_altered + "/aliases.json";
98
99    // デバッグ出力: 検索対象の技名
100    println!(
101        "Finding move: '{}' for character '{}'",
102        character_move, character_arg_altered
103    );
104
105    // aliases.json が存在する場合エイリアス変換処理　結果：エイリアスに対応する実際の技入力を取得
106    if Path::new(&aliases_path).exists() {
107        // ファイル読み込み&デシリアライズ　結果：move_aliases
108        let data_from_file = fs::read_to_string(&aliases_path).map_err(|e| {
109            AppError::FileNotFound(format!(
110                "{}のエイリアスファイル読み込みに失敗しました: {e}",
111                character_arg_altered
112            ))
113        })?;
114
115        let move_aliases =
116            serde_json::from_str::<Vec<MoveAliases>>(&data_from_file).map_err(AppError::Json)?;
117
118        // 各エイリアスと入力を比較　結果：一致した場合は実際の技入力に変換
119        for x_aliases in &move_aliases {
120            for y_aliases in &x_aliases.aliases {
121                if y_aliases.to_lowercase() == character_move.to_lowercase() {
122                    // エイリアスが見つかった場合、対応する技入力に変換
123                    character_move = x_aliases.input.clone();
124                    println!("Alias found: '{}' → '{}'", y_aliases, character_move);
125                    break;
126                }
127            }
128        }
129    }
130
131    // 大文字小文字と空白を区別しない比較のための正規化
132    let normalized_character_move = character_move.to_lowercase().replace(' ', "");
133    println!(
134        "Normalized move input for search: '{}'",
135        normalized_character_move
136    );
137
138    // 技リスト内で検索　結果：該当する技のインデックスを取得
139    for (i, x_move) in moves_info.iter().enumerate() {
140        // 技名が一致した場合インデックス返却
141        if x_move.name.to_lowercase() == character_move.to_lowercase() {
142            println!("Found move by name at index {}: '{}'", i, x_move.name);
143            return Ok(i);
144        }
145    }
146
147    for (i, x_move) in moves_info.iter().enumerate() {
148        // 技入力が一致した場合インデックス返却（正規化して比較）
149        let normalized_move_input = x_move.input.to_lowercase().replace(' ', "");
150        if normalized_move_input == normalized_character_move {
151            println!("Found move by input at index {}: '{}'", i, x_move.input);
152            return Ok(i);
153        }
154    }
155
156    // 技入力の一部一致で検索　結果：部分一致した技のインデックスを取得
157    for (i, x_move) in moves_info.iter().enumerate() {
158        // 技名に入力文字列が含まれる場合インデックス返却
159        if x_move
160            .name
161            .to_lowercase()
162            .contains(&character_move.to_lowercase())
163        {
164            println!(
165                "Found move by partial name match at index {}: '{}'",
166                i, x_move.name
167            );
168            return Ok(i);
169        }
170    }
171
172    for (i, x_move) in moves_info.iter().enumerate() {
173        // 技入力に入力文字列が含まれる場合インデックス返却
174        if x_move
175            .input
176            .to_lowercase()
177            .contains(&character_move.to_lowercase())
178        {
179            println!(
180                "Found move by partial input match at index {}: '{}'",
181                i, x_move.input
182            );
183            return Ok(i);
184        }
185    }
186
187    // エラーメッセージ生成と返却　結果：技が見つからなかった場合エラー返却
188    let error_msg = format!(
189        "Move `{}` was not found for character `{}`!",
190        character_move, character_arg_altered
191    );
192    println!("Error: {}", error_msg);
193    Err(AppError::MoveNotFound(error_msg))
194}

196/// 文字列から括弧内のコンテンツを抽出する関数
197///
198/// # 引数
199/// * `input` - 括弧を含む可能性のある文字列
200///
201/// # 戻り値
202/// 括弧内のコンテンツを含むOption<String>、見つからない場合はNone
203fn extract_bracket_content(input: &str) -> Option<String> {
204    if let Some(start) = input.find('(') {
205        if let Some(end) = input.find(')') {
206            if end > start {
207                return Some(input[start + 1..end].to_string());
208            }
209        }
210    }
211    None
212}

214/// エイリアスから技の入力を検索する非同期関数
215///
216/// # 概要
217/// 指定されたエイリアス（技の別名）から、対応する技の入力コマンドを検索する
218///
219/// # 引数
220/// * `alias` - 技のエイリアス（別名）
221/// * `moves_aliases` - エイリアス情報のスライス
222///
223/// # 戻り値
224/// 技の入力コマンドを含む `Result<String>` を返す
225#[allow(dead_code)]
226pub async fn find_move_by_alias(alias: &str, moves_aliases: &[MoveAliases]) -> Result<String> {
227    // 各エイリアスエントリを走査
228    for move_aliases in moves_aliases {
229        // エイリアス配列を走査して一致確認
230        for a in &move_aliases.aliases {
231            if a.to_lowercase() == alias.to_lowercase() {
232                // 一致したらその技の入力コマンドを返す
233                return Ok(move_aliases.input.clone());
234            }
235        }
236    }
237
238    // 一致するエイリアスが見つからない場合はエラー
239    Err(AppError::MoveNotFound(format!(
240        "技 '{alias}' が見つかりませんでした"
241    )))
242}

275    // テスト実行前に一時的に現在のディレクトリを変更し、テスト後に元に戻す構造体
276    struct TempWorkingDir {
277        original_dir: std::path::PathBuf,
278    }
279
280    impl TempWorkingDir {
281        fn new(dir: &std::path::Path) -> Self {
282            let original_dir = env::current_dir().expect("現在のディレクトリを取得できません");
283            env::set_current_dir(dir).expect("ディレクトリを変更できません");
284            Self { original_dir }
285        }
286    }
287
288    impl Drop for TempWorkingDir {
289        fn drop(&mut self) {
290            env::set_current_dir(&self.original_dir).expect("元のディレクトリに戻れません");
291        }
292    }
