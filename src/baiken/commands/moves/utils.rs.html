9/// 与えられた `MoveInfo` と `MoveAliases` のデータから、
10/// 通常技の情報を文字列として整形して返す非同期関数。
11///
12/// # 引数
13/// * `moves_info` - ムーブ情報のスライス
14/// * `aliases_data` - ムーブのエイリアス情報のスライス
15///
16/// # 戻り値
17/// 通常技の一覧を改行区切りの文字列で返す。
18pub async fn get_normal_moves(moves_info: &[MoveInfo], aliases_data: &[MoveAliases]) -> String {
19    // 通常技情報格納用文字列初期化
20    let mut normal_moves = String::new();
21    // moves_info から通常技のみを抽出（move_type が "normal" であるもの）
22    for moves in moves_info
23        .iter()
24        .take_while(|x| x.move_type.to_lowercase() == "normal")
25    {
26        // 技情報を整形して追加：入力と名称を表示
27        normal_moves = normal_moves.clone() + "\n- **" + &moves.input + "**";
28        // エイリアス情報を確認
29        for moves_aliases in aliases_data {
30            // 一致する入力があればエイリアス情報を追加
31            if moves.input == moves_aliases.input {
32                normal_moves += "\n\t別名 → `";
33                // エイリアス一覧をカンマ区切りで追加
34                for a in 0..moves_aliases.aliases.len() {
35                    if a == moves_aliases.aliases.len() - 1 {
36                        normal_moves = normal_moves.clone() + &moves_aliases.aliases[a];
37                    } else {
38                        normal_moves = normal_moves.clone() + &moves_aliases.aliases[a] + "`, `";
39                    }
40                }
41                normal_moves += "`\n";
42            }
43        }
44    }
45    // 整形済み通常技情報文字列を返却
46    normal_moves
47}

49/// 与えられた `MoveInfo` と `MoveAliases` のデータから、
50/// スペシャル技の情報を文字列として整形して返す非同期関数。
51///
52/// # 引数
53/// * `moves_info` - ムーブ情報のスライス
54/// * `aliases_data` - ムーブのエイリアス情報のスライス
55///
56/// # 戻り値
57/// スペシャル技の一覧を改行区切りの文字列で返す。
58pub async fn get_special_moves(moves_info: &[MoveInfo], aliases_data: &[MoveAliases]) -> String {
59    let mut special_moves = String::new();
60    // 通常技をスキップし、move_type が "special" または "other" の技を対象
61    for moves in moves_info
62        .iter()
63        .skip_while(|x| x.move_type.to_lowercase() == "normal")
64        .take_while(|x| {
65            x.move_type.to_lowercase() == "special" || x.move_type.to_lowercase() == "other"
66        })
67    {
68        special_moves = special_moves.clone() + "\n- **" + &moves.input + "**";
69        for moves_aliases in aliases_data {
70            // 一致する入力があればエイリアス情報を追加
71            if moves.input == moves_aliases.input {
72                special_moves += "\n\t別名 → `";
73                // エイリアス一覧をカンマ区切りで追加
74                for a in 0..moves_aliases.aliases.len() {
75                    if a == moves_aliases.aliases.len() - 1 {
76                        special_moves = special_moves.clone() + &moves_aliases.aliases[a];
77                    } else {
78                        special_moves = special_moves.clone() + &moves_aliases.aliases[a] + "`, `";
79                    }
80                }
81                special_moves = special_moves.clone() + "`\n";
82            }
83        }
84    }
85    special_moves
86}

88/// 与えられた `MoveInfo` と `MoveAliases` のデータから、

98    let mut super_moves = String::new();

101        .iter()

103    {
104        super_moves = super_moves.clone() + "\n- **" + &moves.input + "**";
105        for moves_aliases in aliases_data {
106            // 一致する入力があればエイリアス情報を追加
107            if moves.input == moves_aliases.input {
108                super_moves += "\n\t別名 → `";
109                // エイリアス一覧をカンマ区切りで追加
110                for a in 0..moves_aliases.aliases.len() {
111                    if a == moves_aliases.aliases.len() - 1 {
112                        super_moves = super_moves.clone() + &moves_aliases.aliases[a];
113                    } else {
114                        super_moves = super_moves.clone() + &moves_aliases.aliases[a] + "`, `";
115                    }
116                }
117                super_moves = super_moves.clone() + "`\n";
118            }
119            continue;
120        }
121    }
122    super_moves
