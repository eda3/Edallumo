26/// キャラクターデータを読み込む関数
27///
28/// キャラクター名を受け取り、必要なデータファイルを読み込んでパースします
29///
30/// # 引数
31/// * `character` - ユーザーが入力したキャラクター名
32/// * `ctx` - コマンドコンテキスト
33///
34/// # 戻り値
35/// 成功時は `(キャラクター名, 技情報, エイリアス情報)` のタプル、失敗時は `AppError`
36async fn load_character_data(
37    character: &str,
38    ctx: &Context<'_>,
39) -> Result<(String, Vec<MoveInfo>, Vec<MoveAliases>), AppError> {
40    // キャラクター探索処理　find関数呼出
41    let character_arg_altered = match find::find_character(&character.to_string()).await {
42        Ok(character_arg_altered) => character_arg_altered, // キャラクター名称取得
43        Err(err) => {
44            ctx.say(err.to_string()).await?; // エラーメッセージ送信
45            println!("{}", ("Error: ".to_owned() + &err.to_string()).red()); // エラー出力
46            return Err(AppError::CharacterNotFound(err.to_string())); // エラー時早期終了
47        }
48    };
49
50    // キャラクターファイルパス生成　文字列結合
51    let char_file_path =
52        "data/".to_owned() + &character_arg_altered + "/" + &character_arg_altered + ".json";
53    // キャラクターファイル読み込み　ファイル読み出し
54    let char_file_data = fs::read_to_string(char_file_path)
55        .expect(&("\nFailed to read '".to_owned() + character + ".json" + "' file."));
56
57    // キャラクター情報デシリアライズ　JSONパース
58    let moves_info = serde_json::from_str::<Vec<MoveInfo>>(&char_file_data).unwrap();
59
60    println!(
61        "{}",
62        ("Successfully read '".to_owned() + &character_arg_altered + ".json' file.").green() // 成功出力
63    );
64
65    // エイリアスファイルパス生成　文字列結合
66    let aliases_path = "data/".to_owned() + &character_arg_altered + "/aliases.json";
67    // エイリアスファイル読み込み　ファイル読み出し
68    let aliases_data = fs::read_to_string(&aliases_path)
69        .expect(&("\nFailed to read '".to_owned() + &aliases_path + "' file."));
70
71    // エイリアス情報デシリアライズ　JSONパース
72    let aliases_data = serde_json::from_str::<Vec<MoveAliases>>(&aliases_data).unwrap();
73
74    Ok((character_arg_altered, moves_info, aliases_data))
75}

77/// 指定された技種別に対する埋め込みメッセージを作成する関数
78///
79/// # 引数
80/// * `category` - 技の種別
81/// * `moves_info` - 技情報
82/// * `aliases_data` - エイリアス情報
83/// * `character_arg_altered` - キャラクター名
84///
85/// # 戻り値
86/// 埋め込みメッセージのベクター
87async fn create_embeds_for_move_type(
88    category: &TypeChoice,
89    moves_info: &[MoveInfo],
90    aliases_data: &[MoveAliases],
91    character_arg_altered: &str,
92) -> Vec<CreateEmbed> {
93    let mut vec_embeds = Vec::new(); // 埋め込みメッセージ群格納用ベクター
94
95    // 埋め込みタイトル生成　キャラクター名表示
96    let embed_title =
97        "__**".to_owned() + &character_arg_altered.replace('_', " ") + " Moves / Aliases**__";
98    // 埋め込みURL生成　Dustloop Wiki URL構築
99    let embed_url = "https://dustloop.com/w/GGST/".to_owned() + character_arg_altered + "#Overview";
100    // 埋め込みフッター生成　補足メッセージ
101    let embed_footer = CreateEmbedFooter::new(
102        "Try the \"/help notes\" command for usage notes and specifics.\nOr \"/report\" to request a new aliases."
103    );
104
105    // 技種別に応じた処理分岐
106    match category {
107        TypeChoice::All => {
108            // 通常技取得　ユーティリティ関数呼出
109            let normal_moves = get_normal_moves(moves_info, aliases_data).await;
110            // スペシャル技取得　ユーティリティ関数呼出
111            let special_moves = get_special_moves(moves_info, aliases_data).await;
112            // 必殺技取得　ユーティリティ関数呼出
113            let super_moves = get_super_moves(moves_info, aliases_data).await;
114
115            // 通常技埋め込み作成　CreateEmbed呼出
116            let normals_embed = CreateEmbed::new()
117                .color(EMBED_COLOR) // 埋め込み色設定
118                .title(embed_title.clone()) // タイトル設定
119                .url(embed_url.clone()) // URL設定
120                .description(normal_moves); // 説明文設定
121
122            // スペシャル技埋め込み作成
123            let specials_embed = CreateEmbed::new()
124                .color(EMBED_COLOR)
125                .description(special_moves); // 説明文設定
126
127            // 必殺技埋め込み作成　フッター追加
128            let supers_embed = CreateEmbed::new()
129                .color(EMBED_COLOR)
130                .description(super_moves)
131                .footer(embed_footer);
132
133            vec_embeds.push(normals_embed); // 埋め込み追加
134            vec_embeds.push(specials_embed); // 埋め込み追加
135            vec_embeds.push(supers_embed); // 埋め込み追加
136        }
137        TypeChoice::Normals => {
138            // 通常技取得　ユーティリティ関数呼出
139            let normal_moves = get_normal_moves(moves_info, aliases_data).await;
140
141            let normals_embed = CreateEmbed::new()
142                .color(EMBED_COLOR)
143                .title(embed_title)
144                .url(embed_url)
145                .description(normal_moves)
146                .footer(embed_footer);
147
148            vec_embeds.push(normals_embed); // 埋め込み追加
149        }
150        TypeChoice::Specials => {
151            // スペシャル技取得　ユーティリティ関数呼出
152            let special_moves = get_special_moves(moves_info, aliases_data).await;
153
154            let specials_embed = CreateEmbed::new()
155                .color(EMBED_COLOR)
156                .title(embed_title)
157                .url(embed_url)
158                .description(special_moves)
159                .footer(embed_footer);
160
161            vec_embeds.push(specials_embed); // 埋め込み追加
162        }
163        TypeChoice::Supers => {
164            // 必殺技取得　ユーティリティ関数呼出
165            let super_moves = get_super_moves(moves_info, aliases_data).await;
166
167            let supers_embed = CreateEmbed::new()
168                .color(EMBED_COLOR)
169                .title(embed_title)
170                .url(embed_url)
171                .description(super_moves)
172                .footer(embed_footer);
173
174            vec_embeds.push(supers_embed); // 埋め込み追加
175        }
176    };
177
178    vec_embeds
179}

181/// 技一覧コマンド
182///
183/// 指定したキャラクターの技一覧とエイリアスを表示する
184///
185/// # 引数
186/// * `ctx` - コマンド実行コンテキスト
187/// * `character` - キャラクター指定文字列
188/// * `category` - 技種別選択（"all", "normals", "specials", "supers"）
189/// # 戻り値
190/// `Result<(), Error>` を返す
191#[poise::command(prefix_command, slash_command)]
192pub async fn moves(
193    ctx: Context<'_>, // コマンド実行コンテキスト
194    #[min_length = 2]
195    #[description = "Character name or nickname."] // キャラクター名またはニックネーム
196    character: String, // キャラクター指定文字列
197    #[rename = "type"]
198    #[description = "Move type."] // 技種別指定
199    category: TypeChoice, // 技種別選択値
200) -> Result<(), AppError> {
201    println!(
202        "{}",
203        ("Command Args: '".to_owned() + &character + "'").purple() // 引数出力
204    );
205
206    // 入力チェック実施　条件確認
207    if (check::adaptive_check(
208        ctx,
209        check::CheckOptions::DATA_FOLDER
210            | check::CheckOptions::NICKNAMES_JSON
211            | check::CheckOptions::CHARACTER_FOLDERS
212            | check::CheckOptions::CHARACTER_JSONS,
213    )
214    .await)
215        .is_err()
216    {
217        return Ok(()); // チェック失敗時早期終了
218    }
219
220    // キャラクターデータ読み込み
221    let Ok((character_arg_altered, moves_info, aliases_data)) =
222        load_character_data(&character, &ctx).await
223    else {
224        return Ok(());
225    };
226
227    // 埋め込みメッセージ作成
228    let vec_embeds = create_embeds_for_move_type(
229        &category,
230        &moves_info,
231        &aliases_data,
232        &character_arg_altered,
233    )
234    .await;
235
236    // 返信オブジェクト初期化　CreateReply生成
237    let mut reply = poise::CreateReply::default();
238    reply.embeds.extend(vec_embeds); // 埋め込みメッセージ追加
239
240    ctx.send(reply).await?; // 返信送信
241
242    Ok(()) // 正常終了返却
243}
