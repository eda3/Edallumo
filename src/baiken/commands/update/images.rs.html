1//! # images.rs
2//!
3//! 画像データ更新モジュール。
4//! Dustloopウェブサイトからキャラクターの技画像データを取得し、
5//! ローカルのimages.jsonファイルに保存する機能を提供する。
6
7// 外部クレート読み込み
8extern crate ureq; // HTTPリクエスト用クレート
9
10// 必要なインポート
11use crate::{commands::update::images_json::images_to_json, CHARS}; // 画像JSON変換関数とキャラクター定数
12use colored::Colorize; // ターミナル出力の色付け
13use std::{fs::OpenOptions, time::Instant}; // ファイル操作と時間計測
14
15// 定数定義
16const SITE_LINK: &str = "https://dustloop.com/wiki/api.php?action=cargoquery&format=json&limit=100&tables=MoveData_GGST&fields=MoveData_GGST.input%2C%20MoveData_GGST.name%2C%20MoveData_GGST.images%2C%20MoveData_GGST.hitboxes&where=chara%3D%22"; // Dustloop API リクエスト前半部
17const SITE_HALF: &str =
18    "%22&order_by=MoveData_GGST.type%20ASC%2C%20MoveData_GGST.input%20ASC&utf8=1"; // Dustloop API リクエスト後半部
19
20/// キャラクター画像データ取得関数
21///
22/// # 概要
23/// Dustloopウェブサイトから指定されたキャラクターの技画像データを取得し、
24/// ローカルのimages.jsonファイルに保存する。
25/// 全キャラクターまたは特定のキャラクターを対象に実行可能。
26///
27/// # 引数
28/// * `chars_ids` - キャラクターIDの配列（CHARS.len()サイズ）
29/// * `specific_char` - 特定のキャラクターID（"all"の場合は全キャラクター対象）
30///
31/// # 例
32/// ```rust,no_run
33/// get_char_images(CHARS, "Sol_Badguy").await;
34/// ```
35pub async fn get_char_images(chars_ids: &[&str; CHARS.len()], specific_char: &str) {
36    // 更新時間計測開始
37    let now = Instant::now();
38
39    if specific_char == "all" {
40        // 全キャラクター処理
41        for (x, char_id) in chars_ids.iter().enumerate() {
42            // 処理開始ログ出力
43            println!(
44                "{}",
45                ("Creating ".to_owned() + char_id + " 'images.json' file.").green()
46            );
47
48            // 画像JSONファイルパス生成
49            let images_json_path = "data/".to_owned() + char_id + "/images.json";
50
51            // 複数キャラクターのimages.jsonファイル作成
52            let file = OpenOptions::new()
53                .create(true)
54                .truncate(true)
55                .write(true)
56                .open(images_json_path)
57                .expect(&("\nFailed to open ".to_owned() + char_id + " 'images.json' file."));
58
59            // 画像リクエストリンク生成
60            let character_images_link =
61                SITE_LINK.to_owned() + &char_id.replace('_', " ") + SITE_HALF;
62
63            // Dustloopサイトへリクエスト送信
64            let mut char_images_response_json = ureq::get(&character_images_link).call().unwrap();
65
66            // Dustloopサイトが500エラーを返す場合の再試行処理
67            while char_images_response_json.status() == 500 {
68                char_images_response_json = ureq::get(&character_images_link).call().unwrap();
69            }
70
71            // レスポンスを文字列に変換
72            let char_images_response_json = char_images_response_json.into_string().unwrap();
73
74            // レスポンスを処理してJSONファイルにシリアライズ
75            // char_countはどのJSONファイルが更新に失敗したかを特定するためのカウンター
76            images_to_json(char_images_response_json, &file, x).await;
77        }
78    } else {
79        // 特定キャラクター処理
80        println!(
81            "{}",
82            ("Creating ".to_owned() + specific_char + " 'images.json' file.").green()
83        );
84
85        // 画像JSONファイルパス生成
86        let images_json_path = "data/".to_owned() + specific_char + "/images.json";
87
88        // 単一キャラクターのimages.jsonファイル作成
89        let file = OpenOptions::new()
90            .create(true)
91            .truncate(true)
92            .write(true)
93            .open(images_json_path)
94            .expect(&("\nFailed to open ".to_owned() + specific_char + " 'images.json' file."));
95
96        // リクエストリンク生成
97        let character_link = SITE_LINK.to_owned() + &specific_char.replace('_', " ") + SITE_HALF;
98
99        // Dustloopサイトへリクエスト送信
100        let mut char_images_response_json = ureq::get(&character_link).call().unwrap();
101
102        // Dustloopサイトが500エラーを返す場合の再試行処理
103        while char_images_response_json.status() == 500 {
104            char_images_response_json = ureq::get(&character_link).call().unwrap();
105        }
106
107        // レスポンスを文字列に変換
108        let char_images_response_json = char_images_response_json.into_string().unwrap();
109
110        // レスポンスを処理してJSONファイルにシリアライズ
111        // char_countはどのJSONファイルが更新に失敗したかを特定するためのカウンター
112        images_to_json(char_images_response_json, &file, 0).await;
113    }
114
115    // 経過時間計測と表示
116    let elapsed_time = now.elapsed();
117    println!(
118        "{}",
119        ("Updated in ".to_owned() + &elapsed_time.as_secs().to_string() + " seconds.").yellow()
120    );
121}
