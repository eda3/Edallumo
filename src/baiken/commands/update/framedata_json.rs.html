71/// 与えられた JSON 文字列から不要なタグやエンティティを除去する非同期関数
72///
73/// # 引数
74/// * `char_page_response_json` - 前処理対象の JSON 文字列
75///
76/// # 戻り値
77/// 除去後のクリーンな JSON 文字列
78async fn remove_tags(mut char_page_response_json: String) -> String {
79    char_page_response_json = preprocess::preprocess_json(char_page_response_json);
80
81    // 不要な span タグ（色指定）除去
82    char_page_response_json = char_page_response_json
83        .replace(r#"&lt;span class=&quot;colorful-text-4&quot; &gt;"#, "") // 赤色タグ除去
84        .replace(r#"&lt;span class=&quot;colorful-text-2&quot; &gt;"#, "") // 青色タグ除去
85        .replace(r#"&lt;span class=&quot;colorful-text-3&quot; &gt;"#, "") // 緑色タグ除去
86        .replace(r#"&lt;span class=&quot;colorful-text-1&quot; &gt;"#, "") // 紫色タグ除去
87        .replace(r#"&lt;/span&gt;"#, "")                               // 閉じタグ除去
88        .replace(r#"&lt;br&gt;"#, ", ")                                // 改行タグ置換
89        .replace(r#"&lt;br/&gt;"#, ", ")                               // 改行タグ置換
90        .replace(
91            r#" &lt;span class=&quot;tooltip&quot; &gt;Low Profile&lt;span class=&quot;tooltiptext&quot; style=&quot;&quot;&gt;When a character's hurtbox is entirely beneath an opponent's attack. This can be caused by crouching, certain moves, and being short.&lt;/span&gt;&lt;/span&gt;"#,
92            "",
93        ) // tooltipタグ除去
94        .replace(r#"&#039;"#, "'")    // アポストロフィ置換
95        .replace(r#"&amp;#32;"#, "")   // 不要文字除去
96        .replace(r#"'''"#, "")         // 重複引用符除去
97        .replace(r#"; "#, r#"\n"#)      // セミコロン置換
98        .replace(';', r#"\n"#)       // セミコロン置換
99        .replace(r#"\\"#, ""); // バックスラッシュ除去
100    char_page_response_json // 除去後の文字列返却
101}

103/// 特殊な入力または技名称の場合にスキップするかどうかを判断する関数
104fn should_skip_move(move_data: &Data) -> bool {
105    // 特定の入力 ("j.XX during Homing Jump") の場合、処理スキップ
106    if move_data.title.input.is_some()
107        && *move_data.title.input.as_ref().unwrap() == "j.XX during Homing Jump"
108    {
109        return true;
110    }
111
112    // 特定の技名称の場合、処理スキップ
113    if move_data.title.name.is_some() {
114        let name = move_data.title.name.as_ref().unwrap();
115        if name == "Dash Cancel"
116            || name == "Hoverdash"
117            || name == "Finish Blow"
118            || name == "Flight"
119            || name == "Escape"
120        {
121            return true;
122        }
123    }
124
125    false
126}

128/// 移動データを前処理する関数（入力と名前の補完、キャプションの正規化）
129fn preprocess_move_data(move_data: &mut Data) {
130    // 入力情報が未定義の場合、プレースホルダー "-" を設定
131    if move_data.title.input.is_none() {
132        move_data.title.input = Some("-".to_string());
133    }
134
135    // 技名称が未定義の場合、入力情報を技名称として設定
136    if move_data.title.name.is_none() {
137        move_data.title.name = Some(move_data.title.input.as_ref().unwrap().to_string());
138    }
139
140    // キャプションが "Ground" または "Air" の場合、空文字に置換
141    if move_data.title.caption.is_some()
142        && (move_data.title.caption.as_ref().unwrap() == "Ground"
143            || move_data.title.caption.as_ref().unwrap() == "Air")
144    {
145        move_data.title.caption = Some(String::new());
146    }
147}

149/// 入力名を整形する関数
150fn format_input_name(input: &str, name: &str) -> String {
151    if [
152        "Shooting Time Stretch (Accelerate)",
153        "Shooting Time Stretch (Decelerate)",
154        "Terra Metron",
155        "ステイン",
156    ]
157    .contains(&input)
158    {
159        input.to_string()
160    } else {
161        format!("{name}({input})")
162    }
163}

315/// `移動データからMoveInfo構造体を作成する関数`
316fn create_move_info(move_data: &Data, empty: &str) -> MoveInfo {
317    // 入力名の整形
318    let input_str = move_data.title.input.as_deref().unwrap_or("");
319    let name_str = move_data.title.name.as_deref().unwrap_or("");
320    let input_name = format_input_name(input_str, name_str);
321
322    // 文字列フィールドをまとめて取得
323    let (name, guard, active, on_hit, on_block, level, counter, move_type, invincibility, cancel) =
324        set_string_fields(move_data, empty);
325
326    // 整数型フィールドをまとめて取得
327    let (damage, startup, recovery, wall_damage) = set_integer_fields(move_data);
328
329    // 浮動小数点型フィールドをまとめて取得
330    let (risc_gain, risc_loss, input_tension, chip_ratio) = set_float_fields(move_data);
331
332    // スケーリングはユースケースが単独なので個別に取得
333    let scaling = move_data
334        .title
335        .scaling
336        .as_ref()
337        .and_then(|s| s.parse::<f64>().ok());
338
339    // キャプションと備考は既定値が異なるので個別に取得
340    let caption = move_data.title.caption.as_deref().unwrap_or("").to_string();
341    let notes = move_data.title.notes.as_deref().unwrap_or("").to_string();
342
343    // MoveInfo 構造体へ変換
344    MoveInfo {
345        input: input_name,
346        name,
347        damage,
348        guard,
349        startup,
350        active,
351        recovery,
352        on_hit,
353        on_block,
354        level,
355        counter,
356        move_type,
357        risc_gain,
358        risc_loss,
359        wall_damage,
360        input_tension,
361        chip_ratio,
362        scaling,
363        invincibility,
364        cancel,
365        caption,
366        notes,
367    }
368}

370/// フレームデータをJSON形式に変換するメイン関数
371pub async fn frames_to_json(
372    mut char_page_response_json: String,
373    mut file: &File,
374    char_count: usize,
375) {
376    // "-" をプレースホルダーとして設定
377    let empty = String::from("-");
378
379    // 不要タグ除去処理実施　結果：クリーンな JSON 文字列
380    char_page_response_json = remove_tags(char_page_response_json).await;
381
382    // JSON 文字列を Response 構造体へデシリアライズ　結果：move_data_response 取得
383    let mut move_data_response: Response = serde_json::from_str(&char_page_response_json).unwrap();
384    // 技情報群の可変参照取得　結果：char_move_data
385    let char_move_data = &mut move_data_response.cargoquery;
386    // MoveInfo 変換済みデータ格納用ベクター初期化　結果：vec_processed_moves_info
387    let mut vec_processed_moves_info = Vec::new();
388
389    // 各技情報処理ループ　結果：各技情報の補完と変換
390    for move_data in char_move_data {
391        // 前処理
392        preprocess_move_data(move_data);
393
394        // スキップすべき技かどうかを確認
395        if should_skip_move(move_data) {
396            continue;
397        }
398
399        // MoveInfo構造体の作成と追加
400        let processed_moves_info = create_move_info(move_data, &empty);
401        vec_processed_moves_info.push(processed_moves_info);
402    }
403
404    // 変換済み MoveInfo ベクターを整形済み JSON としてファイルへ書き込み
405    file.write_all(&(serde_json::to_vec_pretty(&vec_processed_moves_info).unwrap()))
406        .expect(&("\nFailed to serialize '".to_owned() + CHARS[char_count] + ".json'."));
