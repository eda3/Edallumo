15/// キャラクターデータを読み込む関数
16///
17/// # 引数
18/// * `character` - ユーザーが入力したキャラクター名
19/// * `ctx` - コマンドコンテキスト
20///
21/// # 戻り値
22/// 成功時は正式なキャラクター名、失敗時はエラー
23async fn load_character_data(character: &str, ctx: &Context<'_>) -> Result<String, AppError> {
24    // キャラクター名の正規化　入力に基づく正式名称の取得
25    let character_arg_altered = match find::find_character(&character.to_string()).await {
26        Ok(character_arg_altered) => character_arg_altered, // 正式名称取得
27        Err(err) => {
28            ctx.say(err.to_string()).await?; // エラーメッセージ送信
29            println!("{}", ("Error: ".to_owned() + &err.to_string()).red());
30            return Err(AppError::CharacterNotFound(err.to_string())); // 処理中断
31        }
32    };
33
34    Ok(character_arg_altered)
35}

37/// キャラクターの技情報を読み込む関数
38///
39/// # 引数
40/// * `character_arg_altered` - 正式なキャラクター名
41///
42/// # 戻り値
43/// 成功時は技情報のベクター
44fn load_moves_info(character_arg_altered: &str) -> Vec<MoveInfo> {
45    // JSONファイルパスの組み立て　キャラクター情報ファイルのパス生成
46    let char_file_path =
47        "data/".to_owned() + character_arg_altered + "/" + character_arg_altered + ".json";
48    // JSONファイルの読み込み　キャラクター情報の取得
49    let char_file_data = fs::read_to_string(char_file_path)
50        .expect(&("\nFailed to read '".to_owned() + character_arg_altered + ".json" + "' file."));
51
52    // JSON文字列のデシリアライズ　技情報の構造体へ変換
53    let moves_info = serde_json::from_str::<Vec<MoveInfo>>(&char_file_data).unwrap();
54
55    // JSON読み込み成功の表示　確認メッセージ出力
56    println!(
57        "{}",
58        ("Successfully read '".to_owned() + character_arg_altered + ".json' file.").green()
59    );
60
61    moves_info
62}

172/// 画像リンクJSONをパースする関数
173///
174/// # 引数
175/// * `image_links` - 画像リンクJSONの文字列
176///
177/// # 戻り値
178/// 画像リンク情報のベクター
179fn parse_image_links(image_links: &str) -> Vec<ImageLinks> {
180    println!("Safely deserializing image links...");
181    let mut image_links_vec = Vec::new();
182
183    // JSONが配列であることを確認
184    if let Ok(serde_json::Value::Array(items)) =
185        serde_json::from_str::<serde_json::Value>(image_links)
186    {
187        // 各要素を個別に処理
188        for item in items {
189            let item_json = serde_json::to_string(&item).unwrap_or_default();
190            match serde_json::from_str::<ImageLinks>(&item_json) {
191                Ok(link) => image_links_vec.push(link),
192                Err(e) => {
193                    // エラーを出力するが、処理は継続
194                    println!("Warning: Failed to deserialize item: {e}");
195                    println!("Skipping problematic item: {item_json}");
196                }
197            }
198        }
199    } else {
200        println!("Error: images.json is not a valid JSON array");
201    }
202
203    println!(
204        "Successfully deserialized {count} image links",
205        count = image_links_vec.len()
206    );
207
208    image_links_vec
209}

211/// 技に対応する画像URLを取得する
212///
213/// # 引数
214/// * `move_data` - 技情報
215/// * `image_links` - 画像リンク情報の配列
216///
217/// # 戻り値
218/// 画像のURL（見つからない場合はデフォルト画像）
219fn get_move_image_url(move_data: &MoveInfo, image_links: &[ImageLinks]) -> String {
220    let mut embed_image = IMAGE_DEFAULT.to_string();
221
222    // 括弧を除去した技名を作成（例：「2d(2d)」→「2d」）
223    let cleaned_input = if move_data.input.contains('(') {
224        move_data
225            .input
226            .split('(')
227            .next()
228            .unwrap_or("")
229            .trim()
230            .to_string()
231    } else {
232        move_data.input.to_string()
233    };
234
235    // 括弧内のコマンドを取得（例：「2d(2d)」→「2d」）
236    let bracket_content = if move_data.input.contains('(') && move_data.input.contains(')') {
237        let start = move_data.input.find('(').unwrap_or(0) + 1;
238        let end = move_data.input.find(')').unwrap_or(move_data.input.len());
239        if start < end {
240            move_data.input[start..end].to_string()
241        } else {
242            String::new()
243        }
244    } else {
245        String::new()
246    };
247
248    // 画像リンクの探索　対象技の画像リンクを検索
249    for img_links in image_links {
250        // 完全一致、括弧を除去した技名との一致、または括弧内のコマンドとの一致
251        if (move_data.input.to_lowercase() == img_links.input.to_lowercase()
252            || (!cleaned_input.is_empty()
253                && cleaned_input.to_lowercase() == img_links.input.to_lowercase())
254            || (!bracket_content.is_empty()
255                && bracket_content.to_lowercase() == img_links.input.to_lowercase()))
256            && !img_links.move_img.is_empty()
257        {
258            embed_image = img_links.move_img.to_string(); // 画像リンク更新
259            break; // 探索終了
260        }
261        // 部分一致（バックアップ）
262        else if img_links
263            .input
264            .to_lowercase()
265            .contains(&move_data.input.to_lowercase())
266            && !img_links.move_img.is_empty()
267        {
268            embed_image = img_links.move_img.to_string(); // 画像リンク更新
269        }
270    }
271
272    embed_image
273}

275/// 技情報と画像データを読み込む関数
276///
277/// # 引数
278/// * `character_arg_altered` - 正式なキャラクター名
279/// * `character_move` - ユーザーが入力した技名
280/// * `ctx` - コマンドコンテキスト
281///
282/// # 戻り値
283/// 成功時は (技情報, 画像URL) のタプル、失敗時はエラー
284async fn find_move_and_images(
285    character_arg_altered: &str,
286    character_move: &str,
287    ctx: &Context<'_>,
288) -> Result<(MoveInfo, String), AppError> {
289    // 技情報の読み込み
290    let moves_info = load_moves_info(character_arg_altered);
291
292    // 画像リンク情報の読み込み
293    let image_links = load_image_links(character_arg_altered);
294
295    // 技インデックス検索
296    let move_index = match find::find_move_index(
297        &character_arg_altered.to_string(),
298        character_move.to_string(),
299        &moves_info,
300    )
301    .await
302    {
303        Ok(index) => index,
304        Err(err) => {
305            ctx.say(err.to_string() + "\nView the moves of a character by executing `/moves`.")
306                .await?;
307            println!("{}", ("Error: ".to_owned() + &err.to_string()).red());
308            return Err(AppError::MoveNotFound(err.to_string()));
309        }
310    };
311
312    // 対象技情報の取得　入力に対応する技データの抽出
313    let move_data = moves_info[move_index].clone();
314
315    // 対象技の読み込み成功の表示　確認メッセージ出力
316    println!(
317        "{}",
318        ("Successfully read move '".to_owned()
319            + &move_data.input
320            + "' in '"
321            + character_arg_altered
322            + ".json' file.")
323            .green()
324    );
325
326    // 技画像URLの取得
327    let embed_image = get_move_image_url(&move_data, &image_links);
328
329    Ok((move_data, embed_image))
330}

332/// 技情報の詳細な埋め込みメッセージを作成する関数

346    // 埋め込みメッセージ群生成用ベクターの初期化

350    // 埋め込みフッターの作成　技に関するキャプションを利用

355        .color(EMBED_COLOR) // 埋め込みカラー設定
356        .title(format!(
357            "{}の{input}",
358            character_arg_altered,
359            input = move_info.input
360        ))
361        .url(embed_url) // URL設定
362        .image(embed_image) // 画像リンク設定
363        .fields(vec![
364            (
365                "ダメージ",
366                &move_info.damage.map_or("-".to_string(), |v| v.to_string()),
367                true,
368            ),
369            ("ガード", &move_info.guard, true),
370            ("無敵", &move_info.invincibility, true),
371            (
372                "始動",
373                &move_info.startup.map_or("-".to_string(), |v| v.to_string()),
374                true,
375            ),
376            ("持続", &move_info.active, true),
377            (
378                "硬直",
379                &move_info
380                    .recovery
381                    .map_or("-".to_string(), |v| v.to_string()),
382                true,
383            ),
384            ("ヒット時", &move_info.on_hit, true),
385            ("ガード時", &move_info.on_block, true),
386            ("カウンター", &move_info.counter, true),
387            ("技レベル", &move_info.level, true),
388            (
389                "リスク増加",
390                &move_info
391                    .risc_gain
392                    .map_or("-".to_string(), |v| v.to_string()),
393                true,
394            ),
395            (
396                "リスク減少",
397                &move_info
398                    .risc_loss
399                    .map_or("-".to_string(), |v| v.to_string()),
400                true,
401            ),
402            (
403                "壁ダメージ",
404                &move_info
405                    .wall_damage
406                    .map_or("-".to_string(), |v| v.to_string()),
407                true,
408            ),
409            (
410                "入力緊張度",
411                &move_info
412                    .input_tension
413                    .map_or("-".to_string(), |v| v.to_string()),
414                true,
415            ),
416            (
417                "チップ比率",
418                &move_info

422            ),
423            (
424                "スケーリング",
425                &move_info.scaling.map_or("-".to_string(), |v| v.to_string()),

427            ),
428        ])
429        .footer(embed_footer); // フッター設定
430
431    // 生成した埋め込みメッセージをベクターに追加
432    vec_embeds.push(embed);

437            .color(EMBED_COLOR) // 埋め込みカラー設定
438            .description(&move_info.notes); // 備考記述設定
439        vec_embeds.push(embed2); // ベクターに追加
440    }
441
442    vec_embeds
443}

445/// 技の詳細情報を埋め込みメッセージで表示するコマンド
446///
447/// # 引数
448/// * `ctx` - コマンドコンテキスト
449/// * `character` - キャラクター名または愛称
450/// * `character_move` - 技名、入力、またはエイリアス
451///
452/// # 戻り値
453/// 処理結果 `Result<(), AppError>`
454#[poise::command(prefix_command, slash_command)]
455pub async fn advanced(
456    ctx: Context<'_>,
457    #[min_length = 2]
458    #[description = "キャラクター名または愛称"]
459    character: String,
460    #[min_length = 2]
461    #[rename = "move"]
462    #[description = "技名、入力、またはエイリアス"]
463    character_move: String,
464) -> Result<(), AppError> {
465    // コマンド引数の表示　引数確認用
466    println!(
467        "{}",
468        ("Command Args: '".to_owned() + &character + ", " + &character_move + "'").purple()
469    );
470
471    // 入力チェックの実施　各種前提条件チェック
472    if (check::adaptive_check(
473        ctx,
474        check::CheckOptions::DATA_FOLDER
475            | check::CheckOptions::NICKNAMES_JSON
476            | check::CheckOptions::CHARACTER_FOLDERS
477            | check::CheckOptions::CHARACTER_JSONS
478            | check::CheckOptions::CHARACTER_IMAGES,
479    )
480    .await)
481        .is_err()
482    {
483        return Ok(());
484    }
485
486    // キャラクターデータ読み込み
487    let Ok(character_arg_altered) = load_character_data(&character, &ctx).await else {
488        return Ok(());
489    };
490
491    // 技情報と画像データ読み込み
492    let Ok((move_info, embed_image)) =
493        find_move_and_images(&character_arg_altered, &character_move, &ctx).await
494    else {
495        return Ok(());
496    };
497
498    // 埋め込みメッセージ作成
499    let vec_embeds = create_advanced_embeds(&move_info, &embed_image, &character_arg_altered);
500
501    // 返信メッセージ用オブジェクト生成　送信用オブジェクトの初期化
502    let mut reply = poise::CreateReply::default();
503    // 生成した埋め込みメッセージ群を返信オブジェクトに追加
504    reply.embeds.extend(vec_embeds);
505    // 返信メッセージの送信　Discordへ送信
506    ctx.send(reply).await?;
507
508    // 正常終了の返却　処理完了
509    Ok(())
510}
