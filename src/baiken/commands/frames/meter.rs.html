19/// 指定ムーブの開始フレーム情報からシンボル文字列生成
20///
21/// # 引数
22/// * `move_info` - ムーブ情報構造体参照
23///
24/// # 戻り値
25/// 開始フレームシンボル文字列
26async fn startup_frames(move_info: &MoveInfo) -> String {
27    // Option<i32>の場合は文字列に変換
28    let startup_str = move_info.startup.map_or("-".to_string(), |v| v.to_string());
29    let startup_vec = sep_frame_vec(&startup_str).await; // 開始フレーム分割結果取得
30    let mut meter_msg = String::new(); // メーター文字列初期化
31                                       // println!("startup_vec: {:?}", startup_vec); // デバッグ出力用
32
33    // 単一エントリかつ空または "-"、または1フレームのみの場合
34    if (startup_vec.len() == 1 && startup_vec[0] == "-")
35        || (startup_vec.len() == 1 && startup_vec[0].parse::<u16>().unwrap() == 1)
36    {
37        meter_msg += "-"; // 単一フレーム表現
38    }
39    // 複数エントリの場合の処理
40    else {
41        let mut startup_bra = false; // 括弧有無判定フラグ初期化
42
43        // 各エントリ処理ループ
44        for x in 0..startup_vec.len() {
45            // 数字エントリの場合
46            if let Ok(num) = startup_vec[x].parse::<u16>() {
47                // 数値-1回分ループ処理
48                for _ in 0..num - 1 {
49                    if startup_bra {
50                        // 括弧内：前エントリとの差分回数緑丸追加
51                        for _ in 0..((startup_vec[x].parse::<u16>().unwrap())
52                            - (startup_vec[x - 2].parse::<u16>()).unwrap())
53                        {
54                            meter_msg += GREEN_CIRCLE; // 括弧内緑丸追加
55                        }
56                        break; // ループ中断
57                    }
58                    meter_msg += GREEN_CIRCLE; // 括弧前：緑丸追加
59                }
60            }
61            // 数字以外のエントリの場合
62            else {
63                // "+"記号処理（末尾直前の場合）
64                if x == startup_vec.len() - 2 && startup_vec[x] == "+" {
65                    if let Ok(num) = startup_vec[x + 1].parse::<u16>() {
66                        // 数字変換試行
67                        if num == 1 {
68                            meter_msg += GREEN_CIRCLE; // 単一数値：緑丸置換
69                        } else {
70                            meter_msg.push_str(GREEN_CIRCLE);
71                            meter_msg.push_str(&startup_vec[x]);
72                            // 複数数値：緑丸＋"+"追加
73                        }
74                    } else {
75                        meter_msg.push_str(&startup_vec[x]); // 数字変換失敗：記号そのまま追加
76                    }
77                }
78                // その他の記号処理
79                else {
80                    meter_msg.push_str(&startup_vec[x]); // 記号追加
81                }
82
83                // 括弧・チルダ判定更新
84                if startup_vec[x] == "[" || startup_vec[x] == "~" {
85                    startup_bra = true; // 括弧開始
86                } else if startup_vec[x] == "]" {
87                    startup_bra = false; // 括弧終了
88                }
89            }
90        }
91    }
92    meter_msg // シンボル文字列返却
93}

95/// 指定ムーブのアクティブフレーム情報からシンボル文字列生成
96///
97/// # 引数
98/// * `move_info` - ムーブ情報構造体参照
99///
100/// # 戻り値
101/// アクティブフレームシンボル文字列
102async fn active_frames(move_info: &MoveInfo) -> String {
103    let active_vec = sep_frame_vec(&move_info.active).await; // アクティブフレーム分割結果取得
104    let mut meter_msg = String::new(); // メーター文字列初期化
105                                       // println!("Active vec: {:?}", active_vec); // デバッグ出力用
106
107    if active_vec.len() == 1 && active_vec[0] == "-" {
108        meter_msg += "-"; // 単一ハイフン表現
109    } else {
110        let mut hit_recovery = false; // 括弧有無判定フラグ初期化
111
112        // 各エントリ処理ループ（参照）
113        for active_vec_string in &active_vec {
114            if let Ok(num) = active_vec_string.parse::<u16>() {
115                // 数値エントリの場合、数値分ループ
116                for _ in 0..num {
117                    if hit_recovery {
118                        meter_msg += BLUE_DIAMOND; // 括弧内：青菱形追加
119                    } else {
120                        meter_msg += RED_SQUARE; // 括弧前：赤四角追加
121                    }
122                }
123            }
124            // 数値以外のエントリの場合
125            else {
126                meter_msg.push_str(active_vec_string); // 記号追加
127                if active_vec_string == "(" {
128                    hit_recovery = true; // 括弧開始
129                } else if active_vec_string == ")" {
130                    hit_recovery = false; // 括弧終了
131                }
132            }
133        }
134    }
135    meter_msg // シンボル文字列返却
136}

138/// 指定ムーブのリカバリーフレーム情報からシンボル文字列生成
139///
140/// # 引数
141/// * `move_info` - ムーブ情報構造体参照
142///
143/// # 戻り値
144/// リカバリーフレームシンボル文字列
145async fn recovery_frames(move_info: &MoveInfo) -> String {
146    // Option<i32>の場合は文字列に変換
147    let recovery_str = move_info
148        .recovery
149        .map_or("-".to_string(), |v| v.to_string());
150    let recovery_vec = sep_frame_vec(&recovery_str).await; // リカバリーフレーム分割結果取得
151    let mut meter_msg = String::new(); // メーター文字列初期化
152
153    if recovery_vec.len() == 1 && recovery_vec[0] == "-" {
154        meter_msg += "-"; // 単一ハイフン表現
155    } else {
156        let mut recovery_tilde = false; // チルダ有無判定フラグ初期化
157
158        // 各エントリ処理ループ（添字利用）
159        for x in 0..recovery_vec.len() {
160            if let Ok(num) = recovery_vec[x].parse::<u16>() {
161                // 数値エントリの場合、数値分ループ
162                for _ in 0..num {
163                    if recovery_tilde {
164                        // チルダ内：前エントリとの差分回数青菱形追加
165                        for _ in 0..((recovery_vec[x].parse::<u16>().unwrap())
166                            - (recovery_vec[x - 2].parse::<u16>()).unwrap())
167                        {
168                            meter_msg += BLUE_DIAMOND; // チルダ内青菱形追加
169                        }
170                        break; // ループ中断
171                    }
172                    meter_msg += BLUE_DIAMOND; // チルダ前：青菱形追加
173                }
174            }
175            // 数値以外のエントリの場合
176            else {
177                meter_msg.push_str(&recovery_vec[x]); // 記号追加
178                                                      // チルダ・括弧判定更新
179                recovery_tilde = recovery_vec[x] == "~" || recovery_vec[x] == "(";
180            }
181        }
182    }
183    meter_msg // シンボル文字列返却
184}

186/// 指定文字列を分割しセパレータを保持したベクター返却
187///
188/// # 引数
189/// * `text` - 分割対象文字列
190///
191/// # 戻り値
192/// 分割結果ベクター（空文字・"total"除去済）
193async fn sep_frame_vec(text: &str) -> Vec<String> {
194    let mut result = Vec::new(); // 分割結果格納ベクター初期化
195    let mut last = 0; // 前回インデックス保持用
196
197    // セパレータ（英数字以外）で分割処理
198    for (index, matched) in text.match_indices(|c: char| !(c.is_alphanumeric())) {
199        if last != index {
200            result.push(text[last..index].to_string()); // 文字列部分抽出
201        }
202        result.push(matched.to_string()); // セパレータ抽出
203        last = index + matched.len(); // インデックス更新
204    }
205    if last < text.len() {
206        result.push(text[last..].to_string()); // 残余部分抽出
207    }
208
209    // 空文字および "total" 削除処理（複数エントリの場合）
210    if result.len() > 1 {
211        'outer: loop {
212            let cur_it_len = result.len(); // 現在長さ保持
213
214            // 各エントリ検査ループ
215            for r in 0..result.len() - 1 {
216                if result[r].to_lowercase() == "total" || result[r].is_empty() || result[r] == " " {
217                    result.remove(r); // 不要項目削除
218                    break; // 削除後ループ再開
219                }
220            }
221
222            if cur_it_len == result.len() {
223                break 'outer; // 安定状態判定
224            }
225        }
226    }
227    result // 分割結果返却
228}

230/// キャラクターデータを読み込む関数
231///
232/// # 引数
233/// * `character` - ユーザーが入力したキャラクター名
234/// * `ctx` - コマンドコンテキスト
235///
236/// # 戻り値
237/// 成功時は正式なキャラクター名、失敗時はエラー
238async fn load_character_data(character: &str, ctx: &Context<'_>) -> Result<String, AppError> {
239    // キャラクター探索処理（エイリアス対応）
240    let character_arg_altered = match find::find_character(&character.to_string()).await {
241        Ok(character_arg_altered) => character_arg_altered, // キャラクター名称確定
242        Err(err) => {
243            ctx.say(err.to_string()).await?; // エラーメッセージ送信
244            println!("{}", ("Error: ".to_owned() + &err.to_string()).red()); // エラー出力
245            return Err(AppError::CharacterNotFound(err.to_string())); // エラー時早期返却
246        }
247    };
248
249    Ok(character_arg_altered)
250}

252/// 技データのJSONファイルを読み込み、技情報を取得する
253///
254/// # 引数
255/// * `character_arg_altered` - 正式なキャラクター名
256/// * `character_move` - ユーザーが入力した技名
257/// * `ctx` - コマンドコンテキスト
258///
259/// # 戻り値
260/// (技情報配列, 技のインデックス)
261async fn load_move_data(
262    character_arg_altered: &str,
263    character_move: &str,
264    ctx: &Context<'_>,
265) -> Result<(Vec<MoveInfo>, usize), AppError> {
266    // キャラクターJSONパス取得
267    let char_file_path =
268        "data/".to_owned() + character_arg_altered + "/" + character_arg_altered + ".json";
269
270    // ファイル読み込み
271    let char_file_data = fs::read_to_string(char_file_path.clone()).map_err(|_| {
272        AppError::FileNotFound(format!(
273            "Failed to read '{character_arg_altered}.json' file."
274        ))
275    })?;
276
277    // JSONデシリアライズ　技データ抽出
278    let moves_info: Vec<MoveInfo> =
279        serde_json::from_str(&char_file_data).map_err(AppError::Json)?;
280
281    // 読み込み成功表示
282    println!(
283        "{}",
284        ("Successfully read '".to_owned() + character_arg_altered + ".json' file.").green()
285    );
286
287    // 技インデックス検索
288    let index = match find::find_move_index(
289        &character_arg_altered.to_string(),
290        character_move.to_string(),
291        &moves_info,
292    )
293    .await
294    {
295        Ok(idx) => idx, // 技インデックス取得
296        Err(err) => {
297            // エラー表示
298            ctx.say(err.to_string() + "\nView the moves of a character by executing `/moves`.")
299                .await?;
300            return Err(AppError::MoveNotFound(err.to_string()));
301        }
302    };
303
304    Ok((moves_info, index))
305}

307/// 技情報から適切な画像URLを検索する
308///
309/// # 引数
310/// * `character_arg_altered` - 正式なキャラクター名
311/// * `selected_move_info` - 選択された技情報
312///
313/// # 戻り値
314/// 画像のURL
315async fn find_move_image(
316    character_arg_altered: &str,
317    selected_move_info: &MoveInfo,
318) -> Result<String, AppError> {
319    // 画像JSONパス
320    let image_json_path = "data/".to_owned() + character_arg_altered + "/images.json";
321
322    // 画像JSONファイル読み込み
323    let image_links = fs::read_to_string(image_json_path).map_err(|_| {
324        AppError::FileNotFound(format!(
325            "Failed to read '{character_arg_altered}' images.json file."
326        ))
327    })?;
328
329    // 画像情報デシリアライズ
330    let image_links =
331        serde_json::from_str::<Vec<ImageLinks>>(&image_links).map_err(AppError::Json)?;
332
333    let mut embed_image = String::new(); // 埋め込み画像初期化
334
335    // 括弧を除去した技名を作成（例：「2d(2d)」→「2d」）
336    let cleaned_input = if selected_move_info.input.contains('(') {
337        selected_move_info
338            .input
339            .split('(')
340            .next()
341            .unwrap_or("")
342            .trim()
343            .to_string()
344    } else {
345        selected_move_info.input.to_string()
346    };
347
348    // 括弧内のコマンドを取得（例：「2d(2d)」→「2d」）
349    let bracket_content =
350        if selected_move_info.input.contains('(') && selected_move_info.input.contains(')') {
351            let start = selected_move_info.input.find('(').unwrap_or(0) + 1;
352            let end = selected_move_info
353                .input
354                .find(')')
355                .unwrap_or(selected_move_info.input.len());
356            if start < end {
357                selected_move_info.input[start..end].to_string()
358            } else {
359                String::new()
360            }
361        } else {
362            String::new()
363        };
364
365    // ムーブ画像送信処理
366    for img_links in &image_links {
367        // 完全一致、括弧を除去した技名との一致、または括弧内のコマンドとの一致
368        if selected_move_info.input.to_lowercase() == img_links.input.to_lowercase()
369            || (!cleaned_input.is_empty()
370                && cleaned_input.to_lowercase() == img_links.input.to_lowercase())
371            || (!bracket_content.is_empty()
372                && bracket_content.to_lowercase() == img_links.input.to_lowercase())
373        {
374            // ヒット判定
375            println!(
376                "{}",
377                ("Successfully read move '".to_owned()
378                    + &selected_move_info.input
379                    + "' in '"
380                    + character_arg_altered
381                    + ".json' file.")
382                    .green()
383            ); // 成功出力
384
385            embed_image = if img_links.move_img.is_empty() {
386                String::from(IMAGE_DEFAULT) // デフォルト画像設定
387            } else {
388                img_links.move_img.clone() // ムーブ画像設定
389            };
390            break; // 検索終了
391        }
392    }
393
394    // 画像が見つからなかった場合、部分一致で再検索
395    if embed_image.is_empty() {
396        for img_links in &image_links {
397            if img_links
398                .input
399                .to_lowercase()
400                .contains(&selected_move_info.input.to_lowercase())
401                && !img_links.move_img.is_empty()
402            {
403                embed_image = img_links.move_img.clone();
404                break;
405            }
406        }
407    }
408
409    // デフォルト画像がセットされていなかった場合
410    if embed_image.is_empty() {
411        embed_image = String::from(IMAGE_DEFAULT);
412    }
413
414    Ok(embed_image)
415}

417/// 技情報と画像を検索する関数
418///
419/// # 引数
420/// * `character_arg_altered` - 正式なキャラクター名
421/// * `character_move` - ユーザーが入力した技名
422/// * `ctx` - コマンドコンテキスト
423///
424/// # 戻り値
425/// (技情報, 画像URL)のタプル
426async fn find_move_and_images(
427    character_arg_altered: &str,
428    character_move: &str,
429    ctx: &Context<'_>,
430    _: &str, // character引数は使用しないため無名化
431) -> Result<(MoveInfo, String), AppError> {
432    // 技データの読み込みと検索
433    let (moves_info, index) = load_move_data(character_arg_altered, character_move, ctx).await?;
434
435    // 選択された技情報の取得
436    let selected_move_info = moves_info[index].clone();
437
438    // 画像の検索
439    let embed_image = find_move_image(character_arg_altered, &selected_move_info).await?;
440
441    Ok((selected_move_info, embed_image))
442}

444/// フレームメーター表示用の埋め込みメッセージを作成する関数
445///
446/// # 引数
447/// * `move_info` - 技情報
448/// * `embed_image` - 埋め込む画像のURL
449/// * `character_arg_altered` - 正式なキャラクター名
450///
451/// # 戻り値
452/// 埋め込みメッセージのベクター
453async fn create_meter_embeds(
454    move_info: &MoveInfo,
455    embed_image: &str,
456    character_arg_altered: &str,
457) -> Vec<CreateEmbed> {
458    // フレームメーター文字列生成処理
459    let mut meter_msg = String::from("`"); // バッククォート開始
460    meter_msg += &startup_frames(move_info).await; // 開始フレーム処理
461    meter_msg += &active_frames(move_info).await; // アクティブフレーム処理
462    meter_msg += &recovery_frames(move_info).await; // リカバリーフレーム処理
463    meter_msg += "`"; // バッククォート終了
464
465    let embed_title = format!("__**{}：{}**__", character_arg_altered, move_info.input); // 埋め込みタイトル生成
466
467    let embed_url = "https://dustloop.com/w/GGST/".to_owned() + character_arg_altered + "#Overview"; // 埋め込みURL生成
468
469    let embed = CreateEmbed::new()
470        .color(EMBED_COLOR) // 埋め込み色設定
471        .title(embed_title) // タイトル設定
472        .url(embed_url) // URL設定
473        .fields(vec![
474            ("Startup", &startup_frames(move_info).await, true), // 開始フレームフィールド
475            ("Active", &active_frames(move_info).await, true),   // アクティブフレームフィールド
476            ("Recovery", &recovery_frames(move_info).await, true), // リカバリーフレームフィールド
477        ])
478        .image(embed_image); // 画像設定
479
480    let embed2 = CreateEmbed::new()
481        .color(EMBED_COLOR) // 埋め込み色設定
482        .description(&meter_msg); // 説明文設定
483
484    vec![embed, embed2] // 埋め込みベクター作成
485}

487/// ムーブのフレームメーターを視覚表示するコマンド処理
488///
489/// # 引数
490/// * `ctx` - コマンド実行コンテキスト
491/// * `character` - キャラクター名またはニックネーム
492/// * `character_move` - ムーブ名・入力またはエイリアス
493///
494/// # 戻り値
495/// 処理結果 `Result<(), AppError>`
496#[poise::command(prefix_command, slash_command)]
497pub async fn meter(
498    ctx: Context<'_>, // コマンドコンテキスト
499    #[min_length = 2]
500    #[description = "Character name or nickname."]
501    character: String, // キャラクター指定文字列
502    #[min_length = 2]
503    #[rename = "move"]
504    #[description = "Move name, input or alias."]
505    character_move: String, // ムーブ指定文字列
506) -> Result<(), AppError> {
507    println!(
508        "{}",
509        ("Command Args: '".to_owned() + &character + ", " + &character_move + "'").purple()
510    ); // コマンド引数出力
511
512    if (check::adaptive_check(
513        ctx,
514        check::CheckOptions::DATA_FOLDER
515            | check::CheckOptions::NICKNAMES_JSON
516            | check::CheckOptions::CHARACTER_FOLDERS
517            | check::CheckOptions::CHARACTER_JSONS
518            | check::CheckOptions::CHARACTER_IMAGES,
519    )
520    .await)
521        .is_err()
522    {
523        return Ok(()); // チェック失敗時早期返却
524    }
525
526    // キャラクターデータ読み込み
527    let Ok(character_arg_altered) = load_character_data(&character, &ctx).await else {
528        return Ok(());
529    };
530
531    // 技情報と画像データ読み込み
532    let Ok((selected_move_info, embed_image)) =
533        find_move_and_images(&character_arg_altered, &character_move, &ctx, &character).await
534    else {
535        return Ok(());
536    };
