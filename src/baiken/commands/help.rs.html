4//! Discordコマンド /help 実装モジュール  
5//! ユーザーが指定したコマンドのヘルプ情報を表示する処理を提供  
6//! 自動補完機能を使用し、ユーザーが入力したコマンド名に対応するヘルプメッセージを送信する

16use colored::Colorize;

18/// ヘルプコマンドの自動補完候補を生成する非同期関数
19///
20/// # 引数
21/// * `_` - コマンド実行時のコンテキスト（今回は使用していない）
22/// * `partial` - ユーザーが入力した部分文字列
23///
24/// # 戻り値
25/// ユーザーの入力にマッチする候補文字列の非同期ストリームを返す
26async fn autocomplete_help<'a>(
27    _: Context<'_>,
28    partial: &'a str,
29) -> impl Stream<Item = String> + 'a {
30    // ヘルプ候補の一覧を列挙したストリームを作成する
31    futures::stream::iter(&[
32        "general",
33        "frames",
34        "hitboxes",
35        "fmeter",
36        "moves",
37        "nicknames",
38        "notes",
39        "specifics",
40        "register",
41        "update",
42        "feedback",
43    ])
44    // ユーザー入力 (partial) にマッチする候補のみフィルタリングする
45    .filter(move |name| {
46        // すべて小文字に変換して比較（大文字小文字を区別しない）
47        futures::future::ready(name.to_lowercase().contains(&partial.to_lowercase()))
48    })
49    // 各候補を String 型に変換して返す
50    .map(|name| (*name).to_string())
51}

53/// ヘルプメッセージを表示するコマンド
54///
55/// ユーザーが指定したオプションに応じて、対応するヘルプメッセージを送信する。
56/// オプションに該当するヘルプが存在しない場合は、エラーメッセージを出力する。
57#[poise::command(prefix_command, slash_command, aliases("?"))]
58pub async fn help(
59    ctx: Context<'_>, // コマンド実行時のコンテキスト。ユーザー情報やチャンネル情報を含む。
60    #[description = "Pick a command to display help for."]
61    #[autocomplete = "autocomplete_help"] // オートコンプリートに先ほど定義した関数を使用
62    option: String, // ユーザーが表示したいヘルプの対象コマンドを指定する文字列
63) -> Result<(), AppError> {
64    // コマンド実行時の引数を紫色でログ出力
65    println!(
66        "{}",
67        ("Command Args: '".to_owned() + &option + "'").purple()
68    );
69
70    // ヘルプメッセージの一時格納用変数
71    let help_message;
72
73    // ユーザーの入力に応じて、対応するヘルプ関数を呼び出す
74    match option.trim() {
75        "feedback" => help_feedback(ctx).await,
76        "fmeter" => help_fmeter(ctx).await,
77        "frames" => help_frames(ctx).await,
78        "general" => help_general(ctx).await,
79        "hitboxes" => help_hitboxes(ctx).await,
80        "moves" => help_moves(ctx).await,
81        "nicknames" => help_nicknames(ctx).await,
82        "notes" => help_notes(ctx).await,
83        "register" => help_register(ctx).await,
84        "specifics" => help_specifics(ctx).await,
85        "update" => help_update(ctx).await,
86        _ => {
87            // 入力に該当するヘルプがない場合、エラーメッセージを生成
88            help_message = "Help for `".to_owned() + &option + "` not found!";
89            // Discord にエラーメッセージを送信
90            ctx.say(&help_message).await?;
91            // エラー内容を赤色でコンソール出力
92            println!("{}", ("Error: ".to_owned() + &help_message).red());
93            return Ok(());
94        }
95    }
96
97    Ok(())
98}

248/// 使用上の注意点を説明するヘルプメッセージを送信する関数
249async fn help_notes(ctx: Context<'_>) {
250    let help_msg = r#"
251__**Usage notes.**__
252
253- **`/` commands can be substituted with direct mentions if preferred.**
254    - Doing so will enable the use of shorthand commands.
255        - Example: `@Baiken f sol 2k` same as `/frames sol 2k`.
256        - Example: `@Baiken h ky 6p` same as `/hitboxes ky 6p`.
257        - Example: `@Baiken m leo` same as `/moves leo`.
258        - Example: `@Baiken a chipp` same as `/aliases chipp`.
259
260- **All searching is case insensitive.**
261    - All names, nicknames, moves and aliases are case agnostic.
262    - Example: `/hitboxes ky dp` = `/hitboxes KY dP`.
263
264- **Character searching.**
265    - Characters can be found either using a part of their name, or any of their nicknames.
266    - Example: `/frames Happy Chaos cs` = `/frames happy cs` = `/frames hc cs`.
267
268- **Move searching.**
269    - Moves can be found either using a part of their name, their input, or any of the aliases that exist.
270        - Example: `/frames Anji Needles` = `/frames Anji 236HP` = `/frames Anji ichi`.
271    - Charged moves can be found with or without the use of `[]`.
272        - Example `/frames may 46S` = `/frames may [4]6S`.
273    - For a fully charged dust attack the alias `5D!` can be used instead.
274        - Example: `/frames chipp 5D!`."#;
275
276    let _ = ctx.say(help_msg).await;
277}

298/// キャラクター固有の仕様について説明するヘルプメッセージを送信する関数
299async fn help_specifics(ctx: Context<'_>) {
300    let help_msg = r#"
301__**Character specifics.**__
302
303- **For normals that have levels like Nagoriyuki.**
304  - Add the level number next to the normal.
305  - For Level 1 `fS`: `/frames nago fs`. 
306  - For Level 2 `fS`: `/frames nago fs2`.
307  - For Level 3 `fS`: `/frames nago fs3`.
308  - If it's a level 1 normal nothing needs to be added since it's the default state.
309
310- **For specials that have levels like Goldlewis.**
311  - Add the level number next to the special.
312  - For Level 1 `Thunderbird`: `/frames gold Drone`.
313  - For Level 2 `Thunderbird`: `/frames gold Drone 2`.
314  - For Level 3 `Thunderbird`: `/frames gold Drone 3`.
315  - The above is not always the case depending on the special move and alias used.
316  - For Level 1 `Thunderbird`: `/frames gold D1`.
317  - For Level 2 `Thunderbird`: `/frames gold D2`.
318  - For Level 3 `Thunderbird`: `/frames gold D3`.
319  - See `/aliases gold` for more info on his aliases.
320
321- **For Testament's different Grave Reaper versions use as shown.**
322  - Regular version: `/frames testament 236S`.
323  - Partially charged version: `/frames testament 236S!`.
324  - Fully charged version: `/frames testament 236S!!`.
325"#;
326
327    let _ = ctx.say(help_msg).await;
328}
